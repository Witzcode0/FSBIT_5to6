#include <stdio.h>
#include <string.h>

int main() {
    char fname[50] = "brijesh";

    char lname[] = "gondaliya";

#include <stdio.h>
#include <string.h>

void swapcase(char s[]){
    // printf("%s", s);
    for(int start = 0; start <strlen(s); start++){
        if (s[start] >= 65 && s[start] <= 90){
            // upper to lower
            printf("%c", s[start]+32);
        }else if(s[start] >= 97 && s[start] <= 122){
            // lower to upper
            printf("%c", s[start]-32);
        }else{
            printf("");
        }
    }
}

int main() {
    char str[50] = "TEchnOLoGY";
    swapcase(str);
    return 0;
}
