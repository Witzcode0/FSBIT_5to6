#include <stdio.h>

int main() {
    int num = 5;
    for(int row = 1; row <= num; row++){
        for(int col = 1; col <= num-row; col++){
            printf("  ");
        }
        for(int col = 1; col <= row; col++){
            printf("* ");
        }
        for(int col = 1; col <= row-1; col++){
            printf("* ");
        }
     
        printf("\n");
    }
    for(int row = 1; row <= num; row++){
        for(int col = 1; col <= row; col++){
            printf("  ");
        }
        for(int col = 1; col <= num-row; col++){
            printf("* ");
        }
        for(int col = 1; col <= num-row-1; col++){
            printf("* ");
        }
        printf("\n");
    }

    return 0;
}

    int num = 5;
    for(int row = 1; row <= num; row++){
        for(int col = 5; col >= row ; col--){
            if (row==col){
                printf("%c ", col+64);
            }else{
                printf("  ");
            }
        }

    }
   

    return 0;
